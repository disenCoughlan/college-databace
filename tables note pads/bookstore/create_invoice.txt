CREATE TABLE PURCHASE_INVOICE
(
	P_INV_NO	VARCHAR(8),
	P_INV_DATE	DATE NOT NULL,
	PUBL_CODE	VARCHAR(3) NOT NULL,
	PRIMARY KEY	(P_INV_NO),
	FOREIGN KEY	(PUBL_CODE) REFERENCES PUBLISHER(PUBL_CODE)
);

	CREATE TABLE SALES_INVOICE
(
	S_INV_NO	VARCHAR(8),
	S_INV_DATE	DATE NOT NULL,
	CUST_CODE	VARCHAR(8) NOT NULL,
	PRIMARY KEY	(S_INV_NO),
	FOREIGN KEY 	(CUST_CODE)REFERENCES CUSTOMER(CUST_CODE)
);
