CREATE TABLE P_INVOICE_LINE
(
	P_INV_NO	VARCHAR(9),
	ISBN		VARCHAR(13),
	QTY		INT(4) NOT NULL,
	UNIT_COST	DECIMAL(6,2) NOT NULL,
	PRIMARY KEY	(P_INV_NO, ISBN),
	FOREIGN KEY	(P_INV_NO)REFERENCES purcheases_invoice(P_INV_NO),
	FOREIGN KEY	(ISBN)REFERENCES book (ISBN)
);
CREATE TABLE S_INVOICE_LINE
(
	S_INV_NO	VARCHAR(9),
	ISBN		VARCHAR(13),
	QTY		INT(4) NOT NULL,
	UNIT_PRICE	DECIMAL(6,2) NOT NULL,
	PRIMARY KEY	(S_INV_NO, ISBN),
	FOREIGN KEY	(S_INV_NO)REFERENCES sales_invoice(S_INV_NO),
	FOREIGN KEY	(ISBN)REFERENCES book (ISBN)
);
